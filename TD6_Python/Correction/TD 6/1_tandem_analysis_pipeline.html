#importation de l'outil
from sklearn.pipeline import Pipeline
#actions = liste de tuples
wkf = Pipeline([('cr',StandardScaler()),
                ('pca',PCA(n_components=2,svd_solver='full')),
                ('cm',KMeans(n_clusters=3,random_state=0))])
