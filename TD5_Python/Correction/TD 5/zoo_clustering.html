import numpy
#fonction indice de Dice au carr
def funDice(col1,col2):
    return(0.5*numpy.sum((col1-col2)**2))

#calcul de la matrice de distance -- Indice de Dice
Dice = numpy.zeros(shape=(M.shape[1],M.shape[1]))

    for k2 in range(MN.shape[1]):
